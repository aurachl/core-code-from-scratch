function uniTotal (string) {
  let total = 0;
      for(let i = 0; i < string.length; i++) 
      {
          for ( let j=0; j < string[i].length; j++ ) 
          {
            let char_code = string[i].charCodeAt(j);
            total += char_code;
          }
      }
  return total;
}
